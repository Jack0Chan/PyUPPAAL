# 使用说明

1.生成xtr文件
	./verifyta.exe ./AVNRT_Fake_GroundTruth.xml -f trace_AVNRT -t 1
		Linux: ./verifyta ./AVNRT_Fake_GroundTruth.xml -f trace_AVNRT -t 1

2.生成if中间表达式
	$env:UPPAAL_COMPILE_ONLY=1
	$PSDefaultParameterValues['Out-File:Encoding'] = 'Default'
	./verifyta AVNRT_Fake_GroundTruth.xml > AVNRT_Fake_GroundTruth.if

3.运行tracer custom
	./程序.exe AVNRT_Fake_GroundTruth.if trace_AVNRT-1.xtr [output.txt]
